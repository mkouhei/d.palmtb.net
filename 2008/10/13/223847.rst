こんなスクリプトでカーネルビルドをしているのだが、

.. code-block:: none

   #!/bin/bash
   
   REVISION=$(date +%Y%m%d.%H%M)
   
   case $1 in
   (snip)
   &#39;menuconfig&#39;)
   	sudo make-kpkg --rootcmd fakeroot clean
   	time make-kpkg --rootcmd fakeroot --config menuconfig --initrd --revision=$REVISION \
   kernel-image kernel_headers modules_image
   	;;
   (snip)
